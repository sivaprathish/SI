# Given values
xbar <- 14.6
mu0 <- 15.4
sigma <- 2.5
n <- 25

# Z-test calculation
z <- (xbar - mu0) / (sigma / sqrt(n))

# Significance level
alpha <- 0.05
zalpha <- qnorm(1 - alpha/2)

# P-value
pval <- 2 * pnorm(z)

# Decision
if(abs(z) < zalpha){
  print("Accept Null Hypothesis")
} else {
  print("Reject Null Hypothesis")
}

# Proportion test
p <- 63/100
P0 <- 0.1726
q <- 1 - P0

z <- (p - P0) / sqrt(P0*q/n)
